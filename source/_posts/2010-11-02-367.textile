jquery代码：
[js]
&lt;script type=&quot;text/javascript&quot;&gt;  
&lt;!--   
$(function(){  
//广告收回  
 var upAdBanner = function(){    
  $(&quot;#ad-banner&quot;).slideUp(800);  
 };  
//广告下拉  
 $(&quot;#ad-banner&quot;).slideDown(800);  
//设置延迟的广告收回，这里是8秒  
 setTimeout(upAdBanner,8000);  
}) --&gt;  
&lt;/script&gt; 
[/js]
